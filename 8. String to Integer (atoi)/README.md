```python
class Solution:
    def myAtoi(self, s: str) -> int:
        INT_MIN, INT_MAX = -2**31, 2**31 - 1

        i, n = 0, len(s)

        # 1) skip leading spaces
        while i < n and s[i] == " ":
            i += 1

        # 2) sign
        sign = 1
        if i < n and s[i] in "+-":
            if s[i] == "-":
                sign = -1
            i += 1

        # 3) digits
        ans = 0
        while i < n and "0" <= s[i] <= "9":
            digit = ord(s[i]) - ord("0")

            # 4) clamp check BEFORE overflow would happen
            if ans > INT_MAX // 10 or (ans == INT_MAX // 10 and digit > INT_MAX % 10):
                return INT_MAX if sign == 1 else INT_MIN

            ans = ans * 10 + digit
            i += 1

        return sign * ans
```
